// Handle incoming messages
async function handleIncomingMessage(message) {
  const chatId = message.from;
  const messageBody = message.body.toLowerCase().trim();

  // Find matching automated response
  const automatedResponse = await db.getAutomatedMessages(chatId);
  
  for (const auto of automatedResponse) {
    if (messageBody.includes(auto.trigger.toLowerCase())) {
      setTimeout(() => {
        whatsappClient.sendMessage(chatId, auto.response);
      }, 500);
      break;  // Para depois de encontrar
    }
  }
}

PROBLEMA: Se alguém enviar "xyz" e só tem resposta para "oi", não responde.


✅ DEPOIS (Responde com "*" + palavra-chave):
───────────────────────────────────────────────────────────

// Handle incoming messages
async function handleIncomingMessage(message) {
  const chatId = message.from;
  const messageBody = message.body.toLowerCase().trim();

  // Find matching automated response
  const automatedResponse = await db.getAutomatedMessages(chatId);
  
  // First, try to find a response that matches the specific trigger
  for (const auto of automatedResponse) {
    if (auto.trigger.toLowerCase() === '*' || 
        auto.trigger.toLowerCase() === 'qualquer' || 
        messageBody.includes(auto.trigger.toLowerCase())) {
      setTimeout(() => {
        whatsappClient.sendMessage(chatId, auto.response);
      }, 500);
      return;  // Sai e responde
    }
  }
  
  // If no specific trigger found, look for a global/default response
  for (const auto of automatedResponse) {
    if (auto.trigger.toLowerCase() === '*' || 
        auto.trigger.toLowerCase() === 'qualquer') {
      setTimeout(() => {
        whatsappClient.sendMessage(chatId, auto.response);
      }, 500);
      return;  // Sai e responde
    }
  }
}

BENEFÍCIO: Agora procura primeiro por palavra-chave específica,
depois por "*" (qualquer coisa).

### 2. Configurar Mensagens Automáticas

1. Após conectar, vá para "Mensagens Automáticas"
2. Preencha:
   - **Chat ID**: Número do WhatsApp (com código de país)
   - **Palavra-chave**: A palavra que dispara a resposta
   - **Resposta**: A mensagem que será enviada automaticamente

3. Clique em "Adicionar Mensagem"

#### 🎯 IMPORTANTE: Responder a Qualquer Mensagem

Para responder **QUALQUER PESSOA** que enviar msg, use `*`:

```
Chat ID:      5511999999999
Palavra-chave: *
Resposta:     Obrigado por sua mensagem!
```

Agora BOT RESPONDERÁ A QUALQUER MENSAGEM! ✨
